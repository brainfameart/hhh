The blob's colour comes from the light's colour tint (erase is // alpha-based, so we modulate brightness via alpha; for coloured lights // we draw a separate coloured pass using blendMode:'add'). // // The result is that unlit areas are darkened to the ambient colour and lit // areas show sprites at their true colour tinted by the light colour — the // same look Unity 2D URP lights produce out of the box. // // In EDITOR mode the lighting is rendered at editor zoom/pan so you can see // the effect while placing lights without entering play mode. // ───────────────────────────────────────────────────────────────────────── private _drawLighting(scene: Scene): void { if (!this._lightLayer || !this._worldRoot) return; const g = this._lightLayer; g.clear(); const lightEntities = scene.query('Light2D', 'Transform'); if (lightEntities.length === 0) { // No lights — disable the layer entirely so sprites render at full colour. this._lightLayer.visible = false; this._lightingActive = false; return; } this._lightLayer.visible = true; this._lightingActive = true; const zoom = this._worldRoot.scale.x || 1; // ── 1. Resolve global / ambient light ───────────────────────────────── // Global lights contribute as an ambient colour that covers the scene. // If multiple global lights exist we blend them additively (like Unity). let ambientR = 0, ambientG = 0, ambientB = 0; let hasGlobal = false; for (const e of lightEntities) { const light = e.getComponent('Light2D') as Light2DComponent; if (light.lightType !== 'global') continue; const [r, g2, b] = this._hexToRGB01(light.color); const i = Math.max(0, light.intensity); ambientR += r * i; ambientG += g2 * i; ambientB += b * i; hasGlobal = true; } if (hasGlobal) { ambientR = Math.min(ambientR, 1); ambientG = Math.min(ambientG, 1); ambientB = Math.min(ambientB, 1); } else { // Default ambient: very dark grey so local lights are visible. ambientR = 0.05; ambientG = 0.05; ambientB = 0.05; } // ── 2. Draw local lights (point, spot, freeform) ──────────────────── // We draw two passes per light to get coloured illumination: // Pass A (erase): punch a white hole in the darkness proportional to // the light's falloff — this reveals the sprite colour. // Pass B (add): tint the revealed area with the light's colour, // blending additively on top of the sprite. // // This separates the "how bright" (erase) from "what colour" (add) so // coloured lights feel physically intuitive without a real shader. for (const e of lightEntities) { const light = e.getComponent('Light2D') as Light2DComponent; if (light.lightType === 'global') continue; const tf = e.getComponent('Transform')!; const wx2 = tf.position.x; // world-space centre const wy2 = tf.position.y; const [lr, lg, lb] = this._hexToRGB01(light.color); const intensity = Math.max(0, Math.min(light.intensity, 4)); // Range is in world units. const a1 = angle + sliceAngle / 2; // Draw a pie-slice wedge g.moveTo(wx, wy); g.arc(wx, wy, r, a0, a1); g.closePath(); (g as any).fill({ color: 0xffffff, alpha, blendMode: 'erase' }); } } // Colour add pass if (lr > 0.02 || lg > 0.02 || lb > 0.02) { for (let i = STEPS; i >= 1; i--) { const t = i / STEPS; const r = range * t; const radialBrightness = Math.pow(1 - t, Math.max(0.5, falloff)); const brightness = radialBrightness * intensity * 0.45; const alpha = Math.min(brightness, 0.55); } } } // ── Sprite sync ────────────────────────────────────────────────────────── private _syncSprites(scene: Scene): void { if (!this._spriteLayer) return; const entities = scene.query('Transform', 'SpriteRenderer');
